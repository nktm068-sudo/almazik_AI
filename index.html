from js import window, document, CustomEvent from pyodide import create_proxy status = document.getElementById("status") btn = document.getElementById("start-btn") status.innerText = "Алмазик готов!" btn.disabled = False def handle_speech(event): text = event.detail.lower() document.getElementById("user-text").innerText = "Вы: " + text if "привет" in text: reply = "Привет! Я Алмазик, твой кристальный друг." elif "как дела" in text: reply = "Всё блестяще! У тебя как?" else: reply = "Я тебя услышал, но пока не знаю, что ответить." window.js_speak(reply) # Подписываемся на событие из JS on_speech = create_proxy(handle_speech) window.addEventListener("processSpeech", on_speech) def on_click(event): status.innerText = "Слушаю..." window.js_listen() btn.addEventListener("click", create_proxy(on_click))
